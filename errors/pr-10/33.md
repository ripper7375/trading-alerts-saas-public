```yaml
- name: Check test results
  run: |
    echo "## Test Summary" >> $GITHUB_STEP_SUMMARY
    echo "" >> $GITHUB_STEP_SUMMARY
    # helper to map result -> emoji and text
    print_status() {
      JOB_NAME=$1
      JOB_RESULT=$2
      case "$JOB_RESULT" in
        success)  EMOJI="✅"; TEXT="success";;
        skipped)  EMOJI="⚠️"; TEXT="skipped";;
        cancelled) EMOJI="⛔"; TEXT="cancelled";;
        failure)  EMOJI="❌"; TEXT="failure";;
        *)        EMOJI="❓"; TEXT="$JOB_RESULT";;
      esac
      echo "- ${EMOJI} ${JOB_NAME}: ${TEXT}" >> $GITHUB_STEP_SUMMARY
    }
    print_status "Unit & Component Tests" "${{ needs.unit-and-component-tests.result }}"
    print_status "Integration Tests" "${{ needs.integration-tests.result }}"
    print_status "Build Check" "${{ needs.build-check.result }}"

- name: Fail if any test failed
  if: |
    needs.unit-and-component-tests.result == 'failure' ||
    needs.integration-tests.result == 'failure' ||
    needs.build-check.result == 'failure'
  run: |
    echo "❌ One or more test jobs failed"
    exit 1
```
